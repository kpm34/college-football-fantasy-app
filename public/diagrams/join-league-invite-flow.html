Invite Validation Process

// Invite validation logic async function validateInvite(token: string) { const invite = await getInviteByToken(token) if (!invite) { throw new Error('Invalid invite link') } if (invite.expires_at && new Date(invite.expires_at) < new Date()) { throw new Error('Invite link has expired') } if (invite.max_uses && invite.uses_count >= invite.max_uses) { throw new Error('Invite link has reached maximum uses') } const league = await getLeague(invite.league_id) if (league.current_teams >= league.max_teams) { throw new Error('League is full') } const membership = await checkMembership(league.id, userId) if (membership) { throw new Error('Already a member of this league') } return { invite, league } }

Validation Checklist

  • Token exists in database
  • Token has not expired
  • Token has uses remaining
  • League has open spots
  • User is not already a member
  • League is accepting new members
  • User is authenticated

Error States & Handling

‚ùå Invalid Token
Token doesn't exist or is malformed. Show user-friendly error and suggest getting a new invite.
‚è∞ Expired Invite
Invite has passed its expiration date. Prompt user to request a new invite from commissioner.
üî¢ Max Uses Reached
Invite has been used maximum times. Commissioner needs to create a new invite link.
üèüÔ∏è League Full
League is at maximum capacity. Show waitlist option or suggest browsing other leagues.

üîí Not Authenticated
User needs to log in first. Store redirect URL and return after authentication.

// Browse public leagues API interface LeagueFilter { conference?: 'SEC' | 'ACC' | 'Big 12' | 'Big Ten' draftType?: 'snake' | 'auction' hasOpenSpots: boolean season: number } async function browsePublicLeagues(filters: LeagueFilter) { const leagues = await db.leagues.find({ isPublic: true, status: 'open', currentTeams: { $lt: '$maxTeams' }, ...filters }) return leagues.map(league => ({ ...league, spotsAvailable: league.maxTeams - league.currentTeams, canJoin: !league.draftStarted })) }
